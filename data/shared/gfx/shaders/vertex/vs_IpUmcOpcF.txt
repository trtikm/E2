#include ../version
#include ../common_vertex_shader
layout(location=BINDING_IN_POSITION) in vec3  in_position;
layout(location=BINDING_OUT_DIFFUSE) out vec4  out_colour;
uniform vec4  DIFFUSE_COLOUR;
uniform mat4  TRANSFORM_MATRIX_TRANSPOSED;
void main() {
    gl_Position = vec4(in_position,1.0f) * TRANSFORM_MATRIX_TRANSPOSED;
    out_colour = DIFFUSE_COLOUR;
}
