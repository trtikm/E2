#include ../version
#include ../bindings
layout(location=BINDING_IN_COLOUR) in vec4  in_colour;
layout(location=BINDING_IN_TEXCOORD0) in vec2  in_texture_coords;
layout(location=BINDING_OUT_COLOUR) out vec4  out_colour;
layout(binding=BINDING_TEXTURE_DIFFUSE) uniform sampler2D diffuse_sampler;
void main() {
    out_colour = mix(in_colour,texture(diffuse_sampler,in_texture_coords),0.5f);
}
